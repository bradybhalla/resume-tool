% arguments: name, contacts
\newcommand{\header}[2]{
  \begin{center}
    \textbf{\Huge #1} \\
    \vspace{2pt}
    {\small #2}
  \end{center}
}

% arguments: icon, contact/link
\newcommand{\contact}[2]{
    #1 \hspace{.5pt} #2
}

% arguments: title/school, company/degree, date, location
\newcommand{\experience}[4]{
  \begin{tabular*}{\textwidth}{l@{\extracolsep{\fill}}r}
    \textbf{#1} $|$ \textit{#2} & #3
  \end{tabular*}
}

% arguments: name, skills
\newcommand{\project}[2]{
  \begin{tabular}{p{\textwidth}}
    \textbf{#1} $|$ \textit{#2}
  \end{tabular}
}

% arguments: citation
\newcommand{\publication}[1]{
  {\footnotesize
    \begin{tabular}{p{\textwidth}}
      \textbullet\ #1
    \end{tabular}
  } \\[2pt]
}

% arguments: text
\renewcommand{\description}[1]{

    \begin{tabular}{@{\hspace{10pt}}p{0.97\textwidth}}

    \end{tabular}
  } \\[2pt]
}

% arguments: text
\newcommand{\plain}[1]{
